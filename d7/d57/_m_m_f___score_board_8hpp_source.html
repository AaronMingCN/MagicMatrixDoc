56 /// @brief 如果改变了则绘制
57 /// @return 是否完成了绘制
58 bool DispScoreChange()
59 {
60 bool r = false;
61 // 如果分数发生改变则开始绘图
62 if ((CurrScoreA != NextScoreA) || (CurrScoreB != NextScoreB)) {
63 CurrScoreA = NextScoreA;

71 /// @brief 执行功能
72 /// @param IDelay 询问等待接口
73 /// @return 执行结果
74 virtual MMFExecR_t Exec(InquireDelay* IDelay)
75 {
76 uint16_t irk = 0; // 定义红外线按键值
77 this->DispScore();
78 while (IDelay->Inquire(irk)) {
79 switch (irk) {
80 case IRK_UP:
81 ++NextScoreA;
82 break;
83 case IRK_DOWN:
84 ++NextScoreB;
85 break;
86 case IRK_LEFT:
87 --NextScoreA;
88 break;
89 case IRK_RIGHT:
90 --NextScoreB;
91 break;
92 case IRK_SET: // 长安Set键归零
93 {
94 bool rset = true; // 默认重置
95 for (int i = 0; i < 10; ++i) { // 等待按键
96 // 如果仍然按着set键则等待
97 if (mmhardware.IRRCode(irk, true) && irk == IRK_SET) {
98 delay(100);
99 } else {
100 rset = false;
101 break;
102 }
103 }
104 if (rset) { // 如果需要重置分数
105 this->NextScoreA = 0;
106 this->NextScoreB = 0;
107 }
108 } break;
109 default:
110 break;
111 }
112 // 如果修改了分数则更新显示并等待
113 if (this->DispScoreChange()) {
114 IDelay->IDelay(200, irk);
115 }
116 }
117
118 return EXECR_OK;
119 }
