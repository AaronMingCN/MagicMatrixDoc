26/// 菜单分类ID和菜单项ID都是8位无符号数,共可表示256 * 256个项目

65 /// @brief 执行条目对应功能
66 /// @param CateID 分类ID
67 /// @param ItemID 条目ID
68 /// @param IDelay 询问等待接口
69 /// @return 执行结果
70 MMFExecR_t ExecItem(uint8_t CateID, uint8_t ItemID, InquireDelay *IDelay) {
71 MMFExecR_t r = EXECR_ERROR; // 定义菜单执行结果，默认为EXECR_ERROR
72 for (auto it : Items) {
73 if (it.CateID == CateID && it.ItemID == ItemID) { // 通过传入的菜单键值在条目列表中查找
74 r = this->FPool->ExecFunc(it.FuncID, IDelay);
75 break;
76 }
77 }
78 return r;
79 }

88 bool r = false;

124 /// @brief 切换菜单位置
125 /// @param IDelay 询问等待接口
126 /// @return 切换结果
127 /// @note
128 /// 切换菜单位置时需要前一个功能响应InquireDelay的退出请求 \n
129 /// 前一个功能退出后由主循环执行当前方法来进行功能切换
130 bool Switch(InquireDelay *IDelay) {
131 bool r = false;
132 // 如果下一个位菜单置存在则更新当前菜单位置，否则将下一位置改为当前位置
133 if (this->ItemExists(this->NextMenuCate, this->NextMenuItem)) {
134 this->CurrMenuCate = this->NextMenuCate;
135 this->CurrMenuItem = this->NextMenuItem;
136 // 将当前菜单位置保存到Config
137 mmconfig.Config[CFG_MENUCATE] = this->CurrMenuCate;
138 mmconfig.Config[CFG_MENUITEM] = this->CurrMenuItem;
139 mmconfig.NeedSave = true; // 标记配置文件需要保存
140 // mmconfig.Save(); // 将当前配置保存，已经放在另外线程中处理
141 this->ExecItem(this->CurrMenuCate, this->CurrMenuItem, IDelay); // 循环执行当前菜单功能
142 r = true;
143 } else {
144 this->NextMenuCate = this->CurrMenuCate;
145 this->NextMenuItem = this->CurrMenuItem;
146 }
147 return r;
148 }
