17/// @brief 命令执行类
18/// @details 根据命令字符串执行对应命令 \n
19/// 命令格式：命令 参数1 参数2 ...
20class MMCommand {
21private:
22public:
23 // 将命令行分割
24 void SplitCMD(String CMDString, std::vector<String>& vcmd)
25 {
26 String t;
27 for (auto c : CMDString) {
28 if (c == ' ' || c == '\0' || c == '\n' || c == '\r') { // 如果遇到空格或字符串结束
29 t.trim(); // 去掉多余的空格
30 if (t.length())
31 vcmd.push_back(t);
32 t = "";
33 } else
34 t += c;
35 }
36 t.trim(); // 如果还有没处理完的数据则继续追加
37 if (t.length())
38 vcmd.push_back(t);
39 }
40
41 /// @brief 执行命令
42 /// @param CMDString 命令行字符串
43 /// @note 命令示例：\n
44 /// MM MENU 0 0 \n
45 /// MM SDT 2024 03 15 00 00 00
46 void Exec(String CMDString)
47 {
48 std::vector<String> vcmd; // 定义分割后的命令
49 this->SplitCMD(CMDString, vcmd);
50 if (vcmd.size()) {
51 if (vcmd[0] == "MM") {
52 if (vcmd.size() > 1 && vcmd[1] == "HELP") { // 显示命令帮助
53 UART_USB.println("---------- samples ----------");
54 UART_USB.println("MM MENU 0 0");
55 UART_USB.println("MM SDT 2024 03 15 00 00 00");
56 UART_USB.println("-----------------------------");
57 } else if (vcmd.size() > 3 && vcmd[1] == "MENU") { // 设置菜单位置格式: MM MENU 0 0
58 mmmenu.NextMenuCate = vcmd[2].toInt();
59 mmmenu.NextMenuItem = vcmd[3].toInt();
60 } else if (vcmd.size() > 7 && vcmd[1] == "SDT") { // 设置时间格式: MM SDT 2024 03 15 00 00 00
61 mmhardware.SetDateTime(vcmd[2].toInt(), vcmd[3].toInt(), vcmd[4].toInt(),
62 vcmd[5].toInt(), vcmd[6].toInt(), vcmd[7].toInt());
63 }
64 }
65#ifdef MMDEBUG
66 UART_USB.println("------------------");
67 for (auto c : vcmd) {
68 UART_USB.println("->" + c);
69 }
70#endif
71 }
72 }
73
74} mmcommand;
