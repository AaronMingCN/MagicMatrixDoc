79 mmscr.SetPixel(c, r, 255, (i * 255 / M_WIDTH), 0);
80 ++r, --c;

118 /// @brief 蛇形填充
119 void SnakeFill()
120 {
121 for (int i = 0; i < M_HEIGHT; ++i) { // 行循环
122 if (i % 2)
123 for (int j = M_WIDTH - 1; j >= 0; --j) {
124 mmscr.SetPixel(j, i, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
125 mmscr.Update();
126 delay(10);
127 }
128 else
129 for (int j = 0; j < M_WIDTH; ++j) {
130 mmscr.SetPixel(j, i, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
131 mmscr.Update();

138 /// @brief 矩形填充
139 void RectFill()
140 {
141 for (int i = 0; i < M_HEIGHT / 2; ++i) {
142 for (int j = i; j < M_WIDTH - i; ++j)
143 mmscr.SetPixel(j, i, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
144 for (int j = i; j < M_HEIGHT - i; ++j)
145 mmscr.SetPixel(i, j, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
146 for (int j = (M_WIDTH - i - 1); j >= i; --j)
147 mmscr.SetPixel(j, M_HEIGHT - i - 1, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
148 for (int j = (M_HEIGHT - i - 1); j >= i; --j)
149 mmscr.SetPixel(M_HEIGHT - i - 1, j, 255, (i * 255 / M_HEIGHT), (j * 255 / M_WIDTH));
150 mmscr.Update();
151 delay(200);
152 }
153 mmhardware.matrix.clear();
