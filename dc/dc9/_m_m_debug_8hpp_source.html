125 /// @brief 测试温湿度显示
126 void TestDHT()
127 {
128 UART_USB.println(F("DHTxx Unified Sensor Example"));
129 // Print temperature sensor details.
130 sensor_t sensor;
131 mmhardware.dht.temperature().getSensor(&sensor);
132 UART_USB.println(F("------------------------------------"));
133 UART_USB.println(F("Temperature Sensor"));
134 UART_USB.print(F("Sensor Type: "));
135 UART_USB.println(sensor.name);
136 UART_USB.print(F("Driver Ver: "));
137 UART_USB.println(sensor.version);
138 UART_USB.print(F("Unique ID: "));
139 UART_USB.println(sensor.sensor_id);
140 UART_USB.print(F("Max Value: "));
141 UART_USB.print(sensor.max_value);
142 UART_USB.println(F("°C"));
143 UART_USB.print(F("Min Value: "));
144 UART_USB.print(sensor.min_value);
145 UART_USB.println(F("°C"));
146 UART_USB.print(F("Resolution: "));
147 UART_USB.print(sensor.resolution);
148 UART_USB.println(F("°C"));
149 UART_USB.println(F("------------------------------------"));
150 // Print humidity sensor details.
151 mmhardware.dht.humidity().getSensor(&sensor);
152 UART_USB.println(F("Humidity Sensor"));
153 UART_USB.print(F("Sensor Type: "));
154 UART_USB.println(sensor.name);
155 UART_USB.print(F("Driver Ver: "));
156 UART_USB.println(sensor.version);
157 UART_USB.print(F("Unique ID: "));
158 UART_USB.println(sensor.sensor_id);
159 UART_USB.print(F("Max Value: "));
160 UART_USB.print(sensor.max_value);
161 UART_USB.println(F("%"));
162 UART_USB.print(F("Min Value: "));
163 UART_USB.print(sensor.min_value);
164 UART_USB.println(F("%"));
165 UART_USB.print(F("Resolution: "));
166 UART_USB.print(sensor.resolution);
167 UART_USB.println(F("%"));
168 UART_USB.println(F("------------------------------------"));
169 // Get temperature event and print its value.
170 sensors_event_t event;
171 mmhardware.dht.temperature().getEvent(&event);
172 if (isnan(event.temperature)) {
173 UART_USB.println(F("Error reading temperature!"));
174 } else {
175 UART_USB.print(F("Temperature: "));
176 UART_USB.print(event.temperature);
177 UART_USB.println(F("°C"));
178 }
179 // Get humidity event and print its value.
180 mmhardware.dht.humidity().getEvent(&event);
181 if (isnan(event.relative_humidity)) {
182 UART_USB.println(F("Error reading humidity!"));
183 } else {
184 UART_USB.print(F("Humidity: "));
185 UART_USB.print(event.relative_humidity);
186 UART_USB.println(F("%"));
187 }
