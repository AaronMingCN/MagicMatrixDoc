35 /// @brief 执行功能
36 /// @param IDelay 等待询问接口
37 /// @return 执行结果
38 virtual MMFExecR_t Exec(InquireDelay* IDelay)
39 {
40 mmconfig.SaveIfNeeded(); // 避免显示时占用sd卡无法保存设置
41 mmscr.FillAll(0xFF, 0xFF, 0xFF); // 填充空白
42 mmscr.Update();
43 mmsd.GetLockSD(); // 获得SD锁
44 if (!SD.begin(PIN_SD_SS)) {
45 UART_USB.println("initialization failed!");
46 while (IDelay->IDelay(1000))
47 ; // 如果打开失败则等待
48 } else {
49 UART_USB.println("initialization done.");
50 File dir = SD.open(DIR_BMPLOOP); // 打开SD卡中的bmp文件夹
51 uint16_t IRRCode; // 如果读取到按键信息则
52 if (dir) {
53 while (IDelay->Inquire(IRRCode)) { // 以此读取文件夹下的bmp文件
54 File entry = dir.openNextFile();
55 if (entry) { // 如果文件打开成功
56 if (!entry.isDirectory()) { // 如果不是文件夹
57 mmsd.DrawBitmapFile(entry); // 则将文件绘制
58 }
59 entry.close(); // 关闭文件
60 } else
61 break;
62
63 if (IDelay->IDelay(Interval, IRRCode, true)) {
64 switch (IRRCode) {
65 case IRK_UP: // 如果是上键增加等待500毫秒
66 this->Interval += 500;
67 delay(IRR_WAITKEYTUP);
68 break;
69
70 case IRK_DOWN: // 如果是下键减少等待500毫秒
71 if (this->Interval > 500)
72 this->Interval -= 500;
73 delay(IRR_WAITKEYTUP);
74 break;
75 }
76 IRRCode = IRK_NONE; // 接受完毕后必须重置IRRCode
77 } else
78 dir.rewindDirectory(); // 返回到文件夹首位置
79 }
80 dir.close();
81 }
82 // SD.end();
83 }
84 mmsd.ReleaseSD(); // 释放SD锁
85 return EXECR_OK;
86 }
87};
