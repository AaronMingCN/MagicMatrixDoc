32/// @brief 查询等待类型(接口)

35/// @note
36/// + 返回为false是需要退出
37/// + 传入的参数是等待的毫秒数,在功能块中应当使用IDelay替代delay()并对返回结果作出响应
38/// + 询问等待的时间当不需要继续执行时可能不传入的时间短
39class InquireDelay {
40public:

46 /// @param IRRCode : 以引用方式返回红外线接收值

51 /// @brief 询问并等待
52 /// @param ms : 等待时长，毫秒
53 /// @return true允许，false不允许

56 delay(ms);

64 /// @param ReturnWhenDecode : 接收到红红外编码是否返回，默认false
65 /// @return true允许，false不允许
66 virtual bool IDelay(unsigned long ms, uint16_t& IRRCode, bool ReturnWhenDecode = false)
