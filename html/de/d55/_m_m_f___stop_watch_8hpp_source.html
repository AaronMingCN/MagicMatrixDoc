69 /// @brief 如果改变了则绘制，避免重复绘图
70 /// @return 是否完成了绘制
71 bool DispIfChange()
72 {
73 bool r = false;
74 // 如果时间发生改变则开始绘图
75 if ((CurrRMinu != NextRMinu) || (CurrRSec != NextRSec) || (CurrR100M != NextR100M)) {
76 CurrRMinu = NextRMinu; // 将新的时间更新，并重新绘制
77 CurrRSec = NextRSec;
78 CurrR100M = NextR100M;
79 this->Disp();
80 r = true;
81 }
82 return r;
83 }

85 /// @brief 准备好显示的时间数据
86 void PrepareTime()
87 {
88 // 剩余毫秒数除以60000得到分钟数
89 this->NextRMinu = this->TickPass / 60000;
90 // 剩余毫秒数除以1000得到秒数，然后对60取模得到秒数
91 this->NextRSec = (this->TickPass / 1000) % 60;
92 // 剩余的100毫秒数,先对1000取模然后再整除100
93 this->NextR100M = (this->TickPass % 1000) / 100;
94 }

96 /// @brief 执行功能
97 /// @param IDelay 询问等待接口
98 /// @return 执行结果
99 virtual MMFExecR_t Exec(InquireDelay* IDelay)
100 {
101 this->Disp();
102 uint16_t irk = 0; // 定义红外线按键值
103 while (IDelay->IDelay(150, irk, true)) {
104 switch (irk) {
105 case IRK_UP:
106 break;
107 case IRK_DOWN:
108 if (!this->Running) { // 如果不在运行中
109 this->TickPass = 0;
110 }
111 break;
112 case IRK_RIGHT:
113 if (!this->Running) {
114 this->Running = true;
115 if (!this->TickPass) { // 如果没有开始，则设置计数时间
116 this->TickStart = millis();
117 } else {
118 this->TickStart = millis() - this->TickPass;
119 }
120 }
121 break;
122 case IRK_LEFT:
123 this->Running = false;
124 break;
125 case IRK_SET: // 按set键开始
126 break;
127 default:
128 break;
129 }
130 // 计算经过的时间
131 if (this->Running)
132 this->TickPass = mmhardware.TickPassed(this->TickStart, millis());
133 this->PrepareTime();
134 this->DispIfChange();
135 }
136 return EXECR_OK;
137 }
