71 /// @brief 计算经过的时间
72 /// @param From 开始时间
73 /// @param To 结束时间
74 /// @return 经过的时间
75 /// @details 计算经过的时间，考虑环绕的情况，毫秒或者微妙
76 unsigned long TickPassed(unsigned long From, unsigned long To)
77 {
78 unsigned long r = 0;
79 if (To >= From)
80 r = (To - From); // 如果没有出现环绕
81 else
82 r = ((unsigned long)(-1) - From) + To; // 如果出现环绕
83 return r;
84 }

101 /// @brief 读取红外遥控编码
102 /// @param[out] Code 读取到的编码
103 /// @param WaitData 是否等待接收到编码
104 /// @return 是否收到编码
105 bool IRRCode(uint16_t& Code, bool WaitData = false)
106 {
107 uint16_t r = false;
108 Code = IRK_NONE;
109 // 如果需要等待数据
110 if (WaitData) {
111 while (!(this->irrecv.decode()))
112 delay(10);
113 r = true;
114 Code = this->irrecv.decodedIRData.command;
115 this->irrecv.resume();
116 } else {
117 if (this->irrecv.decode()) { // 如果红外线读取到数据
118 r = true;
119 Code = this->irrecv.decodedIRData.command;
120 this->irrecv.resume();
121 }
122 }
123 return r;
124 }
