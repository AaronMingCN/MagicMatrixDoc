76 /// @brief 执行功能
77 /// @param IDelay 询问等待接口
78 /// @return 执行结果
79 virtual MMFExecR_t Exec(InquireDelay* IDelay)
80 {
81 mmconfig.SaveIfNeeded(); // 避免显示时占用sd卡无法保存设置
82 mmscr.FillAll(0xFF, 0xFF, 0xFF); // 填充空白
83 mmscr.Update();
84
85 uint16_t FileCount = this->GetFileCount(); // 统计文件的个数
86 uint16_t IRRCode = IRK_NONE; // 如果读取到按键信息则
87 this->DrawBMPByOrder(this->CurrOrder);
88 while (IDelay->Inquire(IRRCode)) { // 以此读取文件夹下的bmp文件
89 if (this->CurrOrder != this->NextOrder) { // 避免重复绘制
90 this->CurrOrder = this->NextOrder;
91 this->DrawBMPByOrder(this->CurrOrder);
92 mmconfig.Config[CFG_DISPBMPIDX] = this->CurrOrder; // 将当前序号写入配置
93 mmconfig.NeedSave = true; // 标记需要保存配置
94 }
95
96 switch (IRRCode) {
97 case IRK_RIGHT: // 如果是右键增加等待增加序号
98 if (this->NextOrder < FileCount - 1)
99 ++(this->NextOrder);
100 else
101 this->NextOrder = 0;
102 IDelay->IDelay(IRR_WAITKEYTUP, IRRCode);
103 break;
104
105 case IRK_LEFT: // 如果是左键减少序号
106 if (this->NextOrder > 0)
107 --(this->NextOrder);
108 else
109 this->NextOrder = FileCount - 1;
110 IDelay->IDelay(IRR_WAITKEYTUP, IRRCode);
111 break;
112 }
113 IRRCode = IRK_NONE; // 接受完毕后必须重置IRRCode
114 }
115
116 return EXECR_OK;
117 }
