158 /// @brief 实现InquireDelay方法
159 /// @param[out] IRRCode 读取到的红外线编码
160 /// @return 询问结果
161 virtual bool Inquire(uint16_t& IRRCode)
162 {
163 this->ProcUART(UART_USB); // 处理USB串口
164 this->ProcUART(UART_BLE); // 处理BLE串口
165
166 // this->CheckPIRR(); // 由于功能块内部需要调用红外线数据，取消此部分多线程处理
167 this->IRRSetMenu(IRRCode);
168 Scheduler.yield(); // 释放资源
169 return (mmmenu.NextMenuItem == mmmenu.CurrMenuItem && mmmenu.NextMenuCate == mmmenu.CurrMenuCate);
170 }

179 /// @brief 实现IDelay方法
180 /// @param ms 等待时间
181 /// @param[out] IRRCode 读取到的红外编码
182 /// @param ReturnWhenDecode 当读取到红外编码是否立刻返回，默认false
183 /// @return 询问结果
184 virtual bool IDelay(unsigned long ms, uint16_t& IRRCode, bool ReturnWhenDecode = false)
185 {
186 bool r = true; // 默认结果为true
187 unsigned long startm = millis(); // 记录函数开始时的运行毫秒数
188 unsigned long pass = 0; // 逝去的毫秒数
189 IRRCode = IRK_NONE;
190 do {
191 delay(10);
192 if (!Inquire(IRRCode)) {
193 r = false;
194 break;
195 }
196 if (IRRCode && ReturnWhenDecode)
197 break;
198 // 过去的时间
199 pass = mmhardware.TickPassed(startm, millis());
200 } while (pass < ms);
201 return r;
202 }
