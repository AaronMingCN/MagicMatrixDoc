46 /// @brief 在MMCanvas指定位置绘制MM字符
47 /// @param x 横向值
48 /// @param y 纵向值
49 /// @param mmc 绘制的MM字符
50 /// @param Canvas 目标画布
51 /// @param Erase 是否擦除之前的内容
52 /// @param color 用于绘制的颜色的地址,为空时默认使用红色
53 /// @param ondraw 在绘制时的回调
54 void DrawMMChar(int16_t x, int16_t y, uint16_t mmc, MMCanvas* Canvas, bool Erase = true, RGB* color = NULL, OnDrawChar ondraw = NULL)
55 {
56 RGB rgb = {};
57 if (color)
58 rgb = *color;
59 else
60 rgb = { 255, 0, 0 };
61
62 for (int16_t i = 0; i < MMCHAR_HEIGHT; ++i) {
63 for (int16_t j = 0; j < MMCHAR_WIDTH; ++j) {
64 mmc <<= 1;
65 if (mmc & 0b1000000000000000) {
66 // a[i + y][j + x] = true;
67 if (ondraw)
68 (*ondraw)(j, i, j + x, i + y, rgb); // 如果定义了绘图时的回调则先调用
69 Canvas->SetPixel(j + x, i + y, rgb);
70 } else if (Erase)
71 Canvas->SetPixel(j + x, i + y, 0, 0, 0);
72 }
73 }
74 }
