81 /// @brief 执行倒计时
82 /// @param IDelay : 传入询问等待接口用于相应功能切换
83 void CountDown(InquireDelay* IDelay)
84 {
85 while (IDelay->IDelay(80)) {
86 unsigned long now = millis(); // 取当前开机时间
87 unsigned long pass = mmhardware.TickPassed(this->LastCount, now); // 计算距离上次计数的时间
88 this->LastCount = now;
89 if (this->RemainMill >= pass)
90 this->RemainMill -= pass;
91 else {
92 this->RemainMill = 0;
93 this->Alert(IDelay);
94 }
95 this->PrepareTime(); // 准备需要显示的时间
96 this->DispRemainChange(); // 显示时间更改
97 }
98 }

100 /// @brief 如果改变了则绘制，避免重复绘图
101 /// @return 是否完成了绘制
102 bool DispRemainChange()
103 {
104 bool r = false;
105 // 如果时间发生改变则开始绘图
106 if ((CurrRMinu != NextRMinu) || (CurrRSec != NextRSec) || (CurrR100M != NextR100M)) {
107 CurrRMinu = NextRMinu; // 将新的时间更新，并重新绘制
108 CurrRSec = NextRSec;
109 CurrR100M = NextR100M;
110 this->DispRemain();
111 r = true;
112 }
113 return r;
114 }

123 /// @brief 执行功能块
124 /// @param IDelay : 询问等待接口
125 /// @return 执行结果
126 virtual MMFExecR_t Exec(InquireDelay* IDelay)
127 {
128 this->DispRemain();
129 const uint32_t MAX_MINU = 999;
130 uint16_t irk = 0; // 定义红外线按键值
131 while (IDelay->Inquire(irk)) {
132 switch (irk) {
133 case IRK_UP:
134 if (NextRMinu < MAX_MINU - 5) {
135 NextRMinu += 5;
136 } else
137 NextRMinu = MAX_MINU;
138 break;
139 case IRK_DOWN:
140 if (NextRMinu >= 5) {
141 NextRMinu -= 5;
142 } else {
143 NextRSec = 0; // 如果分钟已经无法再减少了则将秒置零
144 NextR100M = 0;
145 }
146 break;
147 case IRK_RIGHT:
148 if (NextRMinu < MAX_MINU) {
149 ++NextRMinu;
150 }
151 break;
152 case IRK_LEFT:
153 if (NextRMinu >= 1) {
154 --NextRMinu;
155 } else {
156 NextRSec = 0; // 如果分钟已经无法再减少了则将秒置零
157 NextR100M = 0;
158 }
159 break;
160 case IRK_SET: // 按set键开始
161 this->CalRemainMill();
162 if (this->RemainMill > 0) { // 如果计数的时间大于0则开始倒数
163 this->CountDown(IDelay);
164 }
165 break;
166 default:
167 break;
168 }
169 // 如果剩余时间改变则显示并等待
170 if (this->DispRemainChange()) {
171 IDelay->IDelay(200, irk);
172 }
173 }
174 mmhardware.Beep(false);
175 return EXECR_OK;
176 }

197 }
