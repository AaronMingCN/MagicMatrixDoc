442 /// @brief 设置字符值
443 /// @param c 字符
444 /// @param val 值
445 void SetVal(char c, uint16_t val)
446 {
447 // this->Items.insert(std::pair<char, uint16_t>(c, val));
448 this->Items[c] = val;
449 }
450
451 /// @brief 获得字符值
452 /// @param c 字符
453 /// @param Default 默认值，如果没有找到则返回该值
454 /// @return 如果存在则返回字符值，否则返回
455 uint16_t GetVal(char c, uint16_t Default = MMCHAR_FILL)
456 {
457 uint16_t r = Default; // 默认为全部填充
458 std::map<char, uint16_t>::iterator it = this->Items.find(c);
459 if (it != this->Items.end()) {
460 r = it->second;
461 }
462 return r;
463 }
